\subsection{McEliece}
	\noindent
		Najstarším a pravdepodobne najznámejším code-based kryptosystémom je McElieceov kryptosystém \cite{mceliece1978public}.  Jadro systému tvorí kód $C$ dlky $n$ s
		dimenziou $k$ a minimálnou vzdialenosou $d \geq 2t + 1$, kde $t$ je poèet chıb, ktorı vie kód opravi. Pod¾a pôvodného návrhu sa
		pouívajú Goppove kódy, ku ktorım existuje efektívny dekódovací algoritmus.
	\newline
		Verejnı a súkromnı k¾úè zostrojíme nasledovne. Urèíme generujúcu maticu $G$ s rozmermi $k \times n$ pre kód $C$. 
		Ïalej zvolíme náhodnú binárnu regulárnu maticu $S$ s rozmermi $k \times k$ a permutaènú maticu $P$ s rozmermi
		$n \times n$. Verejnı k¾úè tvorí matica $G' = SGP$ a parameter $t$. Súkromnı k¾úè tvoria matice $S$, $G$, $P$.
		
		\begin{algorithm}[h]
			\caption{McEliece - Algoritmus šifrovania}
			\label{McEliece-encrypt}
			\begin{algorithmic}
				\INPUT Správa $m$ dåky $k$
				\OUTPUT Zašifrovaná správa c
				\State $c' \leftarrow mG'$
				\State K zakódovanej správe pripoèítame náhodnı chybovı vektor s váhou $t$.
				\State $c \leftarrow c' + e$,  $wt(e) = t$
				\State \Return $c$
			\end{algorithmic}		
		\end{algorithm}
		
		\begin{algorithm}[h]
			\caption{McEliece - Algoritmus dešifrovania}
			\label{McEliece-decrypt}
			\begin{algorithmic}
				\INPUT Zašifrovaná správa $c$
				\OUTPUT Otvorená správa $m$
				\State Správu $c$ vynásobíme s $P^{-1}$
				\State $c' \leftarrow cP^{-1} = mSG + eP^{-1}$
				\State $m' \leftarrow Decode(c') = mS$;
				\State $m \leftarrow m'S^{-1}$
				\State \Return $m$
			\end{algorithmic}			
		\end{algorithm}

\subsection{Niederreiter}
	\noindent
		K McElieceovmu kryptosystému existuje variant, ktorı namiesto generujúcej matice $G$ vyuíva kontrolnú maticu $H$. Táto duálna forma je známa ako
		Niederreiterov kryptosystém \cite{bernstein2009post}.  V tomto kryptosystéme sa správa $m$ najskôr transformuje na vektor $m'$ dåky $n$ s Hammingovou váhou $t$. Funkciu, ktorá vykonáva
		túto transformáciu oznaèujeme $\phi_{n,t}(m)$.
		Verejnı k¾úè tvorí matica $H'=SHP$ a parameter $t$.	Matica $S$ je náhodná regulárna binárna matica s rozmermi $(n-k)\times(n-k)$ a 
		$P$ je permutaèná matica s rozmermi $n \times n$ a súkromnı k¾úè tvoria	matice $S$,$H$,$P$. Šifrovanı text sa vypoèíta ako syndróm slova $m'$, $c = H'm'$$^T$
		Na dešifrovanie slova $c$ vlastník súkromného k¾úèa najskôr vynásobí slovo $c$ maticou $S^{-1}$ z¾ava, následne aplikuje dekódovací algoritmus a vısledok vynásobí
		maticou $P^{-1}$ z¾ava. $m = P^{-1}decode(S^{-1}SHPm)$
		
		\begin{algorithm}[h]
			\caption{Niederreiter - Algoritmus šifrovania}
			\label{Niederreiter-encrypt}
			\begin{algorithmic}
				\INPUT Správa $m$
				\OUTPUT Zašifrovaná správa $c$
				\State $m' \leftarrow \phi(m)$, dostaneme chybové slovo dåky $n$ s váhou $t$
				\State $c \leftarrow H'm^{T}$
				\State \Return $c$
			\end{algorithmic}
		\end{algorithm}

	\begin{algorithm}	
		\caption{Niederreiter - Algoritmus dešifrovania}
		\label{Niederreiter-decrypt}
		\begin{algorithmic}
			\INPUT Vektor $c$, ktorı predstavuje šifrovanú správu $m$  
			\OUTPUT Dešifrovaná správa, pôvodné $m$
			\State $c' \leftarrow S^{-1}c$
			\State $e' \leftarrow Decode(c')$
			\State $e \leftarrow P^{-1}e'$
			\State $m \leftarrow \phi^{-1}(e)$
		\end{algorithmic}
	\end{algorithm}
