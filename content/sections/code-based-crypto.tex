	\noindent
		Najstarším a pravdepodobne najznámejším code-based kryptosystémom je McEliecov kryptosystém. Jadro systému tvorí kód $C$ dlky $n$ s

		Verejnı a súkromnı k¾úè zostrojíme nasledovne. Urèíme generujúcu maticu $G$ s rozmermi $k \times n$ pre kód $C$.

		$n \times n$. Verejnı k¾úè tvorí matica $G' = SGP$ a parameter $t$. Súkromnı k¾uè tvoria matice $S$, $G$, $P$.
		
		\begin{algorithm}[h]
			\caption{McEliece - Algoritmus šifrovania}
			\label{McEliece-encrypt}
			\begin{algorithmic}
				\INPUT Správa $m$ dåky $k$
				\OUTPUT Zašifrovaná správa c
				\State $c' \leftarrow mG'$
				\State Ku zakódovanej správe pripoèítame náhodnı chybovı vektor s váhou $t$.
				\State $c \leftarrow c' + e$,  $wt(e) = t$
				\State \Return $c$
			\end{algorithmic}		
		\end{algorithm}
		
		\begin{algorithm}[h]
			\caption{McEliece - Algoritmus dešifrovania}
			\label{McEliece-decrypt}
			\begin{algorithmic}
				\INPUT Zašifrovaná správa $c$
				\OUTPUT Otvorená správa m
				\State Správu $c$ vynásobíme s $P^{-1}$
				\State $c' \leftarrow cP^{-1} = mSG + eP^{-1}$
				\State $m' \leftarrow Decode(c') = mS$;
				\State $m \leftarrow m'S^{-1}$
				\State \Return $m$
			\end{algorithmic}			
		\end{algorithm}

		\begin{algorithm}[h]
			\caption{Niederreiter - Algoritmus šifrovania}
			\label{Niederreiter-encrypt}
			\begin{algorithmic}
				\INPUT Správa $m$
				\OUTPUT Zašifrovaná správa $c$
				\State $m' \leftarrow \phi(m)$, dostaneme chybové slovo dåky $n$ s váhou $t$
				\State $c \leftarrow H'm^{T}$
				\State \Return $c$
			\end{algorithmic}
		\end{algorithm}

	\begin{algorithm}	
		\caption{Niederreiter - Algoritmus dešifrovania}
		\label{Niederreiter-decrypt}
		\begin{algorithmic}
			\INPUT Vektor $c$, ktorı predstavuje šifrovanú správu $m$  
			\OUTPUT Dešifrovaná správa, pôvodné $m$
			\State $c' \leftarrow S^{-1}c$

			\State $e \leftarrow P^{-1}e'$
			\State $m \leftarrow \phi^{-1}(e)$
		\end{algorithmic}
	\end{algorithm}
